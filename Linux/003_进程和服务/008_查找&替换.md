```shell
通常是对输出内容操作
grep		# 查找
	
sed			# 查找、替换、插入
-i			# 操作源文件

sed '[定位符号^|$][操作符a|i|d] [内容]' [file]
a[str]		# 新增，在下一行新增
sed '2a 123'# 在第二行后添加123
i[str]		# 插入，在本行之前插入
d			# 删除
p			# 列印，将某个选择的资料印出

s			# 取代
sed 's/[欲替换内容]/[替换文字]'

find
find [路径][参数][表达式]
# 表达式 +大于 -小于
-name		# 查找指定名称文件
-user		# 查找指定用户拥有的文件
-group		# 查找指定组拥有的文件
-/+mtime n	# 查找在N天前被修改过的文件
-atime n	# 查找在N天前被访问过的文件
-type		
-size		
-...
-exec cmd {} \;	# 跟命令
find /var/ -size +2M -exec ls -l {} \;	# 查找文件并执行命令
```
