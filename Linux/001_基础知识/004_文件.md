```shell
# 文件类型
-rwxr-xr-x	-文件
drwxr-xr-x	d目录
lrwxr-xr-x	l链接(快捷方式)
brwxr-xr-x	b设备文件，提供存储德接口设备
crwxr-xr-x	c设备文件，提供穿行德接口设备--键盘鼠标等

# 文件处理
# 复制文件
cp -r [文件名|目录] [目的目录] # 目录递归

# 创建目录
mkdir -p /a/b/c # 递归创建

# 移动/重命名
mv dir1 dir2

# 查看文件
cat/tac
tac 倒序显示
  -A
  -b 		显示行号，忽略空白行
  -n 		显示行号，空白行也会标记
  -E		显示换行符($)
  -T		显示tab(^I)
  -v 		列出看不出的字符
  
nl			查看文件并开启行号
  -b a|t
  -n ln|rn|nrz
  -w
more		从第一行显示
  空格       翻页
  回车	   下一行
  /			搜索
  :f		显示文件名和行数
less		一页一页翻动
  空格	   翻页
head		从头看
  -n 5 	    查看前面5行
tail        从尾部看
  -n 5      查看后面5行
wc			显示文件的行数，字数，字节
  -c		只显示字节
  -w		只显示字数
  -l  		只显示行数
  
```
